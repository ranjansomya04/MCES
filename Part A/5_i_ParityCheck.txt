	AREA RESET,CODE
	ENTRY
	
	LDR R0,=VAL
	LDR R1,[R0]; Store 32 bit Number
	MOV R2,#32; Counter
	MOV R3,#0; Answer
	MOV R6,#1; Parity 
	LDR R9,=0x40000004
	LDR R4,=0x40000000
	
LOOP
	MOVS R1,R1,LSR #1
	ADDCS R3,R3,#1
	SUB R2,R2,#1
	CMP R2,#0
	BNE LOOP
	MOV R7,R3
	STR R3,[R4]
 
LOOP1
	SUB R7,#2
	CMP R7,#1
	BGT LOOP1
	CMP R7,#1
	BEQ X
	MOV R6,#0
X
	STR R6,[R9]
	
	
STOP B STOP

VAL DCD 0x0000000F
	END
